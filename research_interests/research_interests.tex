\vspace{0.5cm}


\begin{itemize}


\item {\large \it Strongly Correlated Electron Systems.}  
\begin{itemize}
\addtolength{\itemsep}{-\itemsep}   
% separation between items reduced.  Needs to be inside item environ;
% why? 
\item Formal models: Hubbard, Heisenberg, Ising, quantum
rotor. 
\item Transport and optical phenomena in quantum wells and dots.  
\item Kondo physics, heavy fermion materials.  High-Tc
cuprate superconductors.
\item 1D: Luttinger liquid physics, nanotubes \& quantum
wires, bosonization techniques. 
\item Vortex physics in superconductors.  
\item More generally: quantum phase transitions, superconductivity and
magnetism.
\end{itemize}

\item {\large \it Two-Dimensional Systems.}
\begin{itemize}
\addtolength{\itemsep}{-\itemsep}   
\item Kosterlitz-Thouless transitions. 
\item Wigner crystals in semiconductors and on cryogenic liquids.
\item Metal-insulator transition in semiconductor heterojunctions.
\item Quantum Hall physics; in heterojunctions, and in
rotating Bose condensates.
\item Vortex lattices, in superconductors and trapped atomic clouds.
%\item Superfluidity and superconductivity in two dimensions.
\end{itemize}



\item {\large \it  Bose-Einstein Condensation.}
\begin{itemize}
\addtolength{\itemsep}{-\itemsep}   
\item Formalism for weakly interacting Bose gas \\ 
(Field theoretical, variational, RG, and effective-action approaches).
\item BEC of high-density excitons in coupled quantum wells.  
\item Interaction of condensates with photon field.
\item Interference of condensates;  states with multiple
macroscopically occupied modes. 
\end{itemize}



\item {\large \it Trapped Atomic Gases.}
\begin{itemize}
\addtolength{\itemsep}{-\itemsep}   
\item Trapping in \emph{Optical Lattice}.  Superfluid-insulator
transition; Bose-Hubbard model.
\item Trapped fermionic atoms -- quantum degeneracy effects; pairing
\& superfluidity; sympathetic cooling.
\item Light scattering off trapped BEC, Super-radiance.  
\item Vortices in rotating clouds.
\item Multi-component (spinor) condensates --  skyrmions,
interpenetrating vortex lattices.
\end{itemize}


\item {\large \it  Non-equilibrium Phenomena.} 
\begin{itemize}
\addtolength{\itemsep}{-\itemsep}   
\item In general: Applying Keldysh or
Boltzmann-equation formalism to non-equilibrium phenomena.
\item Non-equilibrium effects of laser radiation on excitonic
matter and trapped atomic condensates .
\end{itemize}






\end{itemize}
